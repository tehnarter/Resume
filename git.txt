
  git config --global user.name "Your Name"  -  створення імя власникка репозиторію глобально.

  git config --global user.email "your_email@whatever.com"  -  створення email власникка репозиторію глобально.

  git config --local user.name "Your Name"  -  створення імя власникка репозиторію локально.

  git config --local user.email "your_email@whatever.com"  -  створення email власникка репозиторію локально.

  git init  -  створення  нового репозиторію.

  git status  -  перевірка поточного стану репозиторію.
  
  git add .  -  внесення змін у репориторій.

  git add -A  -  внесення змін у репориторій.
  
  git reset  -  можна зняти індексацію цих змін.
  
  git commit -m "назва коміту"  -  комміт змін.

  git log  -  отримання списку зроблених змін проекту.
 
  git config --list  -  список настрених параметрів git.

  git checkout <імя гілки, комміту>  -  перехід між версіями .
 
  git cat-file -t <hash>  -   вивід останнього комміту.
 
  git cat-file -p  <hash>  -  вивід останнього комміту.

  
  git checkout -b < назва гілки>  -  створення нової гілки і перехід на неї.

  git branch < назва гілки>  -  створення нової гілки репозиторію.

  git branch -m < назва гілки>  -  перейменування гілки.

  git branch -d < назва гілки>  -  видалення гілки.  

  git hist --all  -  перегляд гіллок і їх відмінностей.

  git merge <імя віток>  -  злиття гілок.


        РОБОТА З ВІДДАЛЕНИМИ РЕПОЗИТОРІЯМИ


  git clone <url>  - клонування віддаленого репозиторію в локальний .

  git remote  -  перевірка назви віддаленого репозиторію.

  git remote show origin  -  більш детальна інформація про віддалений репозиторій.
  
  git remote set-url origin <url>  -  зміна URL віддаленого репозиторію.

  git remote add origin <url>  -  привязка до віддаленого репозиторію.

  git remote rm origin <url>  -   відвязка від віддаленого репозиторію.

  git remote remove origin  -   відвязка від віддаленого репозиторію.
  
  git push -u origin main  -  загрузка файлів на віддалений репозиторій.

  git branch -a  -  перегляд гілок відаленого репозиторію.

  git pull  -  витяг змін з віддаленого репозиторію.
